# Define as listas de regiões e áreas (fora do on_startup)
define_region_list = {
	region_list = tupi_undiscover_regions
	regions = {
		upper_peru_region
		peru_region
		la_plata_region
		colombia_region
		central_america_region
	}
}

define_area_list = {
	area_list = tupi_undiscover_areas
	areas = {
		ofaie_area
		mato_grosso_area
		pontal_area
		amazon_area
		guapore_area
		goias_area
	}
}

on_startup = {
	# Parte 1: Processa regiões/áreas específicas
	any_region = {
		region_list = tupi_undiscover_regions
		every_province = {
			every_country = {
				limit = { is_a_coastal_tupi_tribe = yes }
				PREV = { undiscover_country = THIS } # Província remove o país
			}
		}
	}
	
	any_area = {
		area_list = tupi_undiscover_areas
		every_province = {
			every_country = {
				limit = { is_a_coastal_tupi_tribe = yes }
				PREV = { undiscover_country = THIS }
			}
		}
	}
	
	# Parte 2: Processa as regiões oceânicas diretamente
	every_country = {
		limit = { is_a_coastal_tupi_tribe = yes }
		caribbean_sea_region = { undiscover_country = PREV }
		pacific_south_america_region = { undiscover_country = PREV }
	}
}
