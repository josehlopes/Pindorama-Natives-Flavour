 GTC_slot_2 = {
     slot = 2
     generic = no
     ai = yes
     has_country_shield = yes

     potential = {
         tag = GTC
         NOT = { map_setup = map_setup_random }
     }

    gtc_guerrilla_tatics = {
        icon = mission_oceania_assemble_an_army
        position = 1
        required_missions = { }

        trigger = {
            has_mil_advisor = yes
            capital_scope = {
                has_terrain = jungle
            }
        }
        
        effect = {
            add_country_modifier = {
                name = "goitaca_guerrilla_warfare"
                duration = 9125 #25 years
            }
        }
    }

    gtc_found_an_army = {
        icon = mission_native_build_army_mission
        position = 2
        required_missions = { }

        trigger = {
            army_tradition = 20
        }

        effect = {
            add_country_modifier = {
                name = "goitaca_standing_army"
                duration = 27375 #75 years
            }
            hidden_effect = {
                every_owned_province = {
                    every_neighbor_province = {
                        limit = {
                            NOT = { owned_by = ROOT }
                            NOT = { is_permanent_claim = ROOT }
                            NOT = { is_core = ROOT }
                        }
                        add_claim = ROOT
                    }
                }
            }
        }
    }
 }

GTC_slot_3 = { #Tronco da Árvore
    slot = 3
    generic = no
    ai = yes
    has_country_shield = yes

    potential = {
        tag = GTC
        NOT = { map_setup = map_setup_random }
    }

    gtc_sons_of_earth = {
        icon = mission_nah_holy_people
        position = 1
        required_missions = { }

        trigger = {
            num_of_owned_provinces_with = {
                value = 2
                culture = ge
                religion = animism
            }
        }

        effect = {
            add_country_modifier = {
                name = "goitaca_sons_of_earth"
                duration = 9125
            }
        }
    }

    gtc_avoid_foreigners = {
        icon = mission_pow_powhatan_fleet
        position = 2
        required_missions = { gtc_sons_of_earth }

        trigger = {
            europe = { has_discovered = ROOT }
            NOT = {
                any_country = {
                    capital_scope = {
                        continent = europe
                    }
                    alliance_with = ROOT
                }
            }
        }

        effect = {
            add_country_modifier = {
                name = "goitaca_hostile_fame"
                duration = -1
            }
        }
    }

    gtc_self_preservation = {
        icon = mission_ne_palisades
        position = 3
        required_missions = { gtc_avoid_foreigners }

        trigger = {
            ROOT = {
                capital_scope = {
                    has_building = native_palisade
                }
            }
        }

        effect = {
            add_country_modifier = {
                name = "goitaca_self_preservation"
                duration = -1
            }
        }
    }
    
    gtc_unite_the_tribes = {
        icon = mission_unite_home_region
        required_missions = { gtc_self_preservation }
        position = 4

        provinces_to_highlight = {
            provincegroup = ge_culture_provinces
        }

        trigger = {
            ge_culture_provinces = {
                num_of_owned_provinces_with = {
                    value = 10
                }
            }
        }

        effect = {
            ge_culture_provinces = {
				limit = {
					NOT = { owned_by = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
            }
            add_government_reform = tribal_federation
        }   
    
    }
}
