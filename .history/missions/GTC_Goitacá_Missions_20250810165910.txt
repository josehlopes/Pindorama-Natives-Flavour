 GTC_slot_2 = {
     slot = 2
     generic = no
     ai = yes
     has_country_shield = yes

     potential = {
         tag = TUA
         NOT = { map_setup = map_setup_random }
     }

    gtc_guerrilla_tatics = {
        icon = mission_oceania_assemble_an_army
        position = 1
        required_missions = { }

        trigger = {
            has_mil_advisor = yes
            capital_scope = {
                has_terrain = jungle
            }
        }
        
        effect = {
            add_country_modifier = {
                name = "goitaca_guerrilla_warfare"
                duration = 9125 #25 years
            }
        }
    }

    gtc_found_an_army = {
        icon = mission_native_build_army_mission
        position = 2
        required_missions = { gtc_guerrilla_tatics }

        trigger = {
            has_mil_advisor_2 = yes
            army_tradition = 20
        }

        effect = {
            add_country_modifier = {
                name = "goitaca_standing_army"
                duration = 27375 #75 years
            }
            hidden_effect = {
                every_owned_province = {
                    every_neighbor_province = {
                        limit = {
                            NOT = { owned_by = ROOT }
                            NOT = { is_permanent_claim = ROOT }
                            NOT = { is_core = ROOT }
                        }
                        add_claim = ROOT
                    }
                }
            }
        }
    }

    gtc_drive_out_europeans = {
    icon = mission_nah_the_enemy_way
        position = 4
        required_missions = { gtc_found_an_army }
        trigger = {
            OR = {
                is_tribal = no
                has_reform = steppe_horde
            }
            NOT = {
                government = native
            }
            has_institution = feudalism
            custom_trigger_tooltip = {
                tooltip = new_world_remove_euro_toolip
                capital_scope = {
                    region_for_scope_province = {
                        type = all
                        NOT = {
                            owner = {
                                technology_group = western
                            }
                        }
                    }
                }
            }
        }
        effect = {
            add_mil_power = 200
            add_country_modifier = {
                name = "military_victory"
                duration = 7300
            }
            custom_tooltip = perma_claim_neighbor_area
            hidden_effect = {
                every_owned_province = {
                    every_neighbor_province = {
                        area = {
                            limit = {
                                NOT = { owned_by = ROOT }
                                NOT = { is_permanent_claim = ROOT    }
                                NOT = { is_core = ROOT    }
                            }
                            add_permanent_claim = ROOT
                        }
                    }
                }
            }
        }
    }
}
